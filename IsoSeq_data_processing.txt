# Pool mapped .bam files for each genotype to create concensus transcriptome
# Also perform the isoseq collapse step (collapse redundant isoforms)

#!/bin/bash -l
#SBATCH --job-name=pool_collapse
#SBATCH --cpus-per-task=4
#SBATCH --mem=80G
#SBATCH --time=24:00:00
#SBATCH -p epyc
#SBATCH --output=logs/pool_analysis_%j.out
#SBATCH --error=logs/pool_analysis_%j.err
#SBATCH --mail-user=wcoll010@ucr.edu
#SBATCH --mail-type=ALL

set -euo pipefail

# 1. Environment Setup
source /opt/linux/rhel/8.x/x86_64/pkgs/miniconda3/py39_4.12.0/etc/profile.d/conda.sh
conda activate "$HOME/.conda/envs/isoseq_run_env"

# 2. Define Directories - Keeping it all in shared
BASE_DIR="/rhome/wcoll010/shared/pacbio/IsoSeq_analyses_2"
INPUT_DIR="${BASE_DIR}/clustered_samples"
OUT_DIR="${BASE_DIR}/pooled_results"
mkdir -p "$OUT_DIR"

# Your 8 Rootstock groups
ROOTSTOCKS=("SO4" "cab33" "1616" "ripa" "sw" "3309" "99R" "1103")

for STOCK in "${ROOTSTOCKS[@]}"; do
    echo "=========================================="
    echo "Processing Rootstock: $STOCK"
    echo "=========================================="

    # Find the MAPPED bam files in the nested folders
    # Example: clustered_samples/cab33_con_R1/cab33_con_R1_mapped.bam
    BAMS=$(ls ${INPUT_DIR}/${STOCK}_*/${STOCK}_*_mapped.bam 2>/dev/null || true)
    
    if [[ -z "$BAMS" ]]; then
        echo "WARNING: No mapped BAM files found for $STOCK. Skipping..."
        continue
    fi

    POOLED_BAM="${OUT_DIR}/${STOCK}_pooled.sorted.bam"
    
    # 3. Merge and Sort
    # Merges individual sample alignments into one master file for the rootstock
    echo "Merging mapped BAMs for $STOCK..."
    samtools merge -@ 3 - $BAMS | \
    samtools sort -@ 3 -m 18G -o "$POOLED_BAM"
    samtools index "$POOLED_BAM"

    # 4. Collapse (This is where the GFF is created)
    # This step reduces redundant reads into unique transcript models
    echo "Running isoseq collapse to create GFF for $STOCK..."
    isoseq collapse --do-not-collapse-extra-5exons \
        --num-threads "${SLURM_CPUS_PER_TASK}" \
        "$POOLED_BAM" \
        "${OUT_DIR}/${STOCK}_pooled.gff"

    echo "Done with $STOCK"
done

echo "Workflow complete. Final GFFs are in $OUT_DIR"

import pandas as pd
import argparse
import sys

def filter_replicate_matrix(counts_file, classification_file, output_file):
    print(f"Loading Master File: {counts_file}")
    
    # 1. Load the list of IDs that survived the SQANTI filter
    class_df = pd.read_csv(classification_file, sep="\t")
    valid_ids = set(class_df['isoform'])
    print(f"  Valid isoforms in filtered list: {len(valid_ids)}")

    # 2. Load the Master abundance file
    df = pd.read_csv(counts_file, sep="\t")
    id_col = df.columns[0]
    
    # 3. Match logic: Split the pipe and check if the short ID is in valid_ids
    # This keeps all your columns (con_R1, inf_R1, etc.) exactly as they are.
    initial_rows = len(df)
    
    # We create a temporary column to do the matching so we don't lose the original ID format if you want to keep it
    # OR we replace it with the short ID for better R compatibility (recommended)
    df[id_col] = df[id_col].str.split('|').str[0]
    
    # Filter
    filtered_df = df[df[id_col].isin(valid_ids)].copy()
    
    print(f"  Rows before filter: {initial_rows}")
    print(f"  Rows after filter:  {len(filtered_df)}")
    print(f"  Saving DE-ready matrix to: {output_file}")
    
    filtered_df.to_csv(output_file, sep="\t", index=False)

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--counts")
    parser.add_argument("--class_file")
    parser.add_argument("--output")
    args = parser.parse_args()
    filter_replicate_matrix(args.counts, args.class_file, args.output)
EOF
